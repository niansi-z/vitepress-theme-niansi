## 设置语言 <Badge text="重要" type="danger" />

在非使用默认 `en-US` 的情况下，在 `.vitepress/config.ts` 中设置 `lang`。

::: code-group

```ts [单语言]
// .vitepress/config.ts
import { defineConfig } from 'vitepress-theme-niansi'

export default defineConfig({
  lang: 'zh-CN',
  // ...
})
```

```ts [多语言]
// .vitepress/config.ts
import { defineConfig } from 'vitepress-theme-niansi'

export default defineConfig({
  locales: {
    root: { label: '简体中文', lang: 'zh-Hans' },
    en: { label: 'English', lang: 'en-US' }
  }
  // ...
})
```

:::
